授業課題5/18

function sample3(event){
let node = document.getElementById('sample3');// サンプル3のIDを持つ要素をnodeに代入しておく
let str = '';// 初期化として空白を代入しておく
for(i = 0; i < 3; ++i)
{// 普通のforループ．iが0から始まり，回るごとに++iで1つづつ増えていく．１，２と増えて，３になったタイミングで条件式 (i< 3)よりループを抜ける
str += 'どうかな？';//ループ文はiが0から2まで回るため，最終的にstrに'どうかな？'が３回足される．
}
// for(attribute in event) { str += ' ' + attribute + ': ' + event[attribute]; }
let newNode = document.createElement('li');// 'li'(箇条書き)を持つ要素をnewNodeとして宣言
let textNode = document.createTextNode(str);// 上のfor文で作成したstrをtextNodeとして宣言
newNode.appendChild(textNode);// 上で宣言したnewNodeにtextNodeを子ノードとして追加する(これで箇条書き文の中に"どうかな？"３連発が追加される)
node.appendChild(newNode);一番上で代入したサンプル3のIDを持つ要素に対し，一つ上のnewNodeを子ノードとして追加する(これで箇条書き文とテキストを実際に見ることができるようになる)
}

ここをクリックするとcssのrule setがclass名の書き換えによって切り替わるよ！

document.getElementsByClassNameが配列で取得されるというのにしばらく気づかずスタックしていました．たしかにElementsなので配列ですね．
